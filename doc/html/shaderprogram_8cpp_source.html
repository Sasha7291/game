5 ShaderProgram::ShaderProgram(const std::string& vertexShader, const std::string& fragmentShader)
6 {
7 compiled = false;
8 id = 0;
9
10 GLuint vertexShaderId = 0;
11 GLuint fragmentShaderId = 0;
12
13 if (!createShader(vertexShader, GL_VERTEX_SHADER, vertexShaderId))
14 {
15 std::cerr << "ERROR::Vertex shader compile time error" << std::endl;
16 return;
17 }
18
19 if (!createShader(fragmentShader, GL_FRAGMENT_SHADER, fragmentShaderId))
20 {
21 std::cerr << "ERROR::Fragment shader compile time error" << std::endl;
22 glDeleteShader(vertexShaderId);
23 return;
24 }
25
26 id = glCreateProgram();
27 glAttachShader(id, vertexShaderId);
28 glAttachShader(id, fragmentShaderId);
29 glLinkProgram(id);
30
31 GLint linked = GL_TRUE;
32 glGetProgramiv(id, GL_LINK_STATUS, &linked);
33 if (linked == GL_FALSE)
34 {
35 GLchar log[1024];
36 glGetProgramInfoLog(id, 1024, nullptr, log);
37 std::cerr << "ERROR::Link shader time error:" << std::endl << log << std::endl;
38 return;
39 }
40 else
41 {
42 compiled = true;
43 glDeleteShader(vertexShaderId);
44 glDeleteShader(fragmentShaderId);
45 }
46 }

290 {
