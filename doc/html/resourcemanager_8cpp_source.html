95std::shared_ptr<Renderer::Texture2d> ResourceManager::loadTextureAtlas(const std::string& textureName,
96 const std::string& texturePath,
97 const std::vector<std::string>& subTextures,
98 const unsigned int subTextureWidth,
99 const unsigned int subTextureHeight)
100{
101 auto texture = loadTexture(textureName, texturePath);
102
103 if (texture != nullptr)
104 {
105 const unsigned int textureWidth = texture->getWidth();
106 const unsigned int textureHeight = texture->getHeight();
107 unsigned int currentTextureOffsetX = 0;
108 unsigned int currentTextureOffsetY = textureHeight;
109
110 for (const auto& currentSubTextureName : subTextures)
111 {
112 glm::vec2 leftBottomUV(static_cast<float>(currentTextureOffsetX) / textureWidth,
113 static_cast<float>(currentTextureOffsetY - subTextureHeight) / textureHeight);
114 glm::vec2 rightTopUV(static_cast<float>(currentTextureOffsetX + subTextureWidth) / textureWidth,
115 static_cast<float>(currentTextureOffsetY) / textureHeight);
116
117 texture->addSubTexture(currentSubTextureName, leftBottomUV, rightTopUV);
118
119 currentTextureOffsetX += subTextureWidth;
120 if (currentTextureOffsetX >= textureWidth)
121 {
122 currentTextureOffsetX = 0;
123 currentTextureOffsetY -= subTextureHeight;
124 }
125 }
126 }
127
128 return texture;
